# ============================================
# BUILD AND RUN
# ============================================

# Clone repository
git clone https://github.com/yourusername/PayalBot.git
cd PayalBot

# Build with Maven
mvn clean package

# Run the bot
java -jar target/PayalBot-1.0.0.jar

# ============================================
# DOCKER BUILD AND RUN
# ============================================

# Build Docker image
docker build -t payal-bot .

# Run Docker container
docker run -d \
  -e BOT_TOKEN=${BOT_TOKEN} \
  -e GEMINI_API_KEY=${GEMINI_API_KEY} \
  -p 8080:8080 \
  --name payal-bot \
  payal-bot
